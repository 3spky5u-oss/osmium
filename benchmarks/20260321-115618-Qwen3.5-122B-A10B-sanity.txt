    // Base case: length 0 or 1 is already sorted
    if arr.len() <= 1 {
        return;
    }

    // Choose the last element as the pivot
    let pivot_idx = arr.len() - 1;
    let pivot = arr[pivot_idx];

    // Partition logic (Lomuto scheme)
    let mut i = 0;
    for j in 0..pivot_idx {
        if arr[j] <= pivot {
            arr.swap(i, j);
            i += 1;
